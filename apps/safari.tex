\section{Bash Startup Script}

#!/bin/bash

# Safari Browser Startup Script
# Purpose: Launch Safari with specified configurations

SAFARI_PATH="/Applications/Safari.app/Contents/MacOS/Safari"
DEFAULT_URL="https://www.apple.com"

# Function to open Safari
open_safari() {
        local url="${1:-$DEFAULT_URL}"
        
        if [ -x "$SAFARI_PATH" ]; then
                open -a Safari "$url"
                echo "Safari launched with URL: $url"
        else
                echo "Error: Safari not found at $SAFARI_PATH"
                exit 1
        fi
}

# Main execution
main() {
        open_safari "$@"
}

main "$@"
\end{lstlisting}
